#include <iostream>

using namespace std;

int main()
{
	int arr[] = { 64, 34, 25, 12, 22, 11, 90 };
	int n = sizeof arr / sizeof arr[0];


	cout << "Unsorted array: ";

	for (int i = 0; i < n; i++)
		cout << arr[i] << ' ';
	cout << endl;



	// Bubble sort

	bool swapped;
	int temp;
	for (int i = 0; i < n - 1; i++)
	{
		swapped = false;
		for (int j = 0; j < n - i - 1; j++)
		{
			if (arr[j] > arr[j + 1])
			{
				temp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = temp;

				swapped = true;
			}
		}

		if (swapped == false)
			break;
	}



	cout << "Sorted array: ";

	for (int i = 0; i < n; i++)
		cout << arr[i] << ' ';
	cout << endl;





	system("pause");
}
