#include <iostream>

using namespace std;

int main()
{
	int arr[] = { 64, 34, 25, 12, 22, 11, 90 };
	int n = sizeof arr / sizeof arr[0];


	// Linear Search
	int find = 13;
	bool isFind = false;

	for (int i = 0; i < n; i++)
	{
		if (arr[i] == find) {
			isFind = true;
			break;
		}
	}

	isFind ? cout << "Found\n" : cout << "Can not find\n";


	system("pause");
}
