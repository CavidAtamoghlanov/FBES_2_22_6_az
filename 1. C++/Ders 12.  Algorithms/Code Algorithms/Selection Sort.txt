#include <iostream>

using namespace std;

int main()
{
	int arr[] = { 64, 25, 12, 22, 11 };
	int n = sizeof arr / sizeof arr[0];


	cout << "Unsorted array: ";

	for (int i = 0; i < n; i++)
		cout << arr[i] << ' ';
	cout << endl;



	// Selection sort

	int min_idx, temp;

	for (int i = 0; i < n - 1; i++)
	{
		min_idx = i;
		for (int j = i + 1; j < n; j++)
			if (arr[j] < arr[min_idx])
				min_idx = j;

		temp = arr[min_idx];
		arr[min_idx] = arr[i];
		arr[i] = temp;
	}



	cout << "Sorted array: ";

	for (int i = 0; i < n; i++)
		cout << arr[i] << ' ';
	cout << endl;





	system("pause");
}
