	printArray(arr, n);

inline bool callBack(int a, int b, bool isAsc) {
	return isAsc ? a > b : b > a;
}

void bubbleSort(int arr[], const int n, bool isAsc) {
	bool swapped;
	int temp;

	for (int i = 0; i < n - 1; i++)
	{
		swapped = false;
		for (int j = 0; j < n - i - 1; j++)
		{
			
			if (callBack(arr[j], arr[j + 1], isAsc))
			{
				temp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = temp;

				swapped = true;
			}
		}

		if (swapped == false)
			break;
	}

}


void printArray(int array[], const int n) {
	for (int i = 0; i < n; i++)
		cout << array[i] << ' ';
	cout << endl;
}
