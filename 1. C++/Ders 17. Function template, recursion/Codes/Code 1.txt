#include <iostream>

using namespace std;


void display(int arr[], const int size1, const int size2) {

	for (int i = 0; i < size1; i++)
	{
		for (int j = 0; j < size2; j++)
		{
			cout << arr[i * size2 + j] << ' ';
		}
		cout << endl;
	}
}

int main()
{
	const int size1 = 3;
	const int size2 = 3;

	int arr[size1][size2]{ 
		{1, 2, 3}, 
		{6 ,7, 8},
		{61 ,72, 8},
	};

	display(arr[0], size1, size2);
}
