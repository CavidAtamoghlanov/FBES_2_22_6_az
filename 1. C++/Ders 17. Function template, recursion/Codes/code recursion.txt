#include <iostream>

using namespace std;

void sayHi(int n) {
	cout << "Hello\n";

	if (n == 1) return;

	sayHi(n - 1);
}

int factorial(int n) { //1
	if (n == 1) return 1;

	return n * factorial(n - 1);
}

void showOddNumbers(int i, int n) {

	if (i == n) return;

}


int sumArrayElements(int arr[], int n) {
	if (n == 1) return arr[0];

	return arr[n - 1] + sumArrayElements(arr, n - 1);
}
