#include<iostream>

using namespace std;


void fillArray(int* arr, int size)
{
	for (int i = 0; i < size; i++)
		//*(arr + i) = rand() % 30;
		arr[i] = rand() % 30;
	

}

void printArray(int* arr, int size)
{
	for (int i = 0; i < size; i++)
		//cout << *(arr + i) << " ";
		cout << arr[i] << " ";
	cout << endl;
}


int* appendArray(int* arr, int *size, int* numeData)
{
	int* newArray = new int[*size + 1];
	for (int i = 0; i < *size; i++)
		newArray[i] = arr[i];
	
	newArray[*size] = *numeData;
	delete[] arr;
	arr = nullptr;

	(*size)++;

	return newArray;

}



void main()
{
	srand(time(NULL));

	/*
		1. by value    => (copy)
		2. reference   => (self => Ozu)
		3. pointer     => (address => copy)

		4. reference pointer => (address ozu gedir)
	*/
	int size = 10;
	int* arr = new int[size] {};

	fillArray(arr, size); 
	printArray(arr, size);
	int x;
	for (size_t i = 0; i < 10; i++)
	{
		x = rand() % 10;

		arr = appendArray(arr, &size, &x);
	}




	printArray(arr, size);

	
}
