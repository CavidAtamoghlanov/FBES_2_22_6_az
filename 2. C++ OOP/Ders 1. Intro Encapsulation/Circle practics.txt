#include <iostream>
#include <cassert>
using namespace std;


class Circle
{
private:
	int _radius;
	const float PI = 3.14;
public:
	Circle() {
		this->_radius = 0;
	}

	Circle(int radius) {
		setRadius(radius);
	}


	//------
	int getRaius() {
		return this->_radius;
	}

	void setRadius(int radius)
	{
		if (radius >= 0) {
			this->_radius = radius;
			return;
		}
		assert(!"Error");
	}

	double getLenght() {
		return 2 * this->PI * this->getRaius();
	}

	double getArea() {
		return this->PI * this->_radius * this->getRaius();
	}
   
};



int main()
{

	Circle cevre1(10);
	cout << cevre1.getArea() << endl;

}
