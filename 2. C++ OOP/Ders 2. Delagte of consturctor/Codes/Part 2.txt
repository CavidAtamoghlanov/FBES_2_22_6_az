class Student {
private:
	char* _name;
	int _age;

public:

	explicit Student() {
		_name = nullptr;
		_age = 0;
	}
	explicit Student(const char*name) {
		int len = strlen(name) + 1;
		strcpy_s(this->_name, len, name);
		_age = 0;
	}

	explicit Student(int age) {
		_name = nullptr;
		_age = age;
	}

	Student(const char* name, int age) {
		int len = strlen(name) + 1;
		strcpy_s(this->_name, len, name);
		_age = age;
	}

	// print
	void print() {
		cout << "Name: " << (_name ? _name : "null") << endl;
		cout << "Age: " << _age << endl;
	}
	
};

int main()
{
	Student s1;
	Student s2("Amin");
	Student s3(15);
	Student s4("Rustem", 11);
