void addNumber(int*& arr, int& size, const int& number) {
	int* newArr = new int[size + 1];

	for (size_t i = 0; i < size; i++)
		newArr[i] = arr[i];

	newArr[size++] = number;

	delete[] arr;
	arr = newArr;
}
